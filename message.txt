    <script>
      /* JavaScript */
      const yesButton = document.getElementById("yes");
      const noButton = document.getElementById("no");
      const body = document.getElementsByTagName("body")[0];

      yesButton.addEventListener("click", function() {
        window.location.href = "https://www.youtube.com/watch?v=HhoATZ1Imtw";
      });

      noButton.addEventListener("click", function() {
        noButton.classList.add("hidden");

        const newButtonContainer = document.createElement("div");
        newButtonContainer.classList.add("button-container");
        newButtonContainer.appendChild(noButton);

        body.appendChild(newButtonContainer);

        const screenWidth = window.innerWidth;
        const screenHeight = window.innerHeight;

        const randomX = Math.floor(Math.random() * (screenWidth - 100));
        const randomY = Math.floor(Math.random() * (screenHeight - 100));

        noButton.style.position = "absolute";
        noButton.style.left = randomX + "px";
        noButton.style.top = randomY + "px";

        noButton.classList.remove("hidden");
      });
    </script>
